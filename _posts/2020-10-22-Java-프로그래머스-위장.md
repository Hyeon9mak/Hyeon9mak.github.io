import java.util.HashMap;

class Solution {
    public int solution(String[][] clothes) {

        for(int i=0; i<clothes.length; i++){

            } else {
                map.put(key, map.get(key) + 1);
            }
        }
        Iterator<Integer> it = map.values().iterator();
        while(it.hasNext()) {
            answer *= it.next().intValue()+1;
        }
        return answer-1;
    }
}
